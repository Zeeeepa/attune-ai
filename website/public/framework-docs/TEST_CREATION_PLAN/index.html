    Testing Strategy¶

    Test Types by Priority¶

    1. Unit Tests (All files)
    2. Test individual functions/methods
    3. Mock external dependencies

    4. Fast execution (<1s per test)

    5. Integration Tests (Core modules)

    6. Test component interactions
    7. Use real dependencies where safe

    8. Moderate execution (<10s per test)

    9. Security Tests (File I/O, user input handling)

    10. Path traversal prevention
    11. Input validation
    12. Exception handling

class Test{ClassName}:
    """Tests for {ClassName}."""

    @pytest.fixture
    def instance(self):
        """Create test instance."""
        return {ClassName}()

    def test_basic_functionality(self, instance):
        """Test basic operation."""
        result = instance.method()
        assert result is not None

    def test_error_handling(self, instance):
        """Test error cases."""
        with pytest.raises(ValueError):
            instance.method(invalid_input)

    def test_edge_cases(self, instance):
        """Test edge cases."""
        assert instance.method(None) == expected
        assert instance.method([]) == expected
